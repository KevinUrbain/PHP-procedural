# Étape 18 — JSON et formats de données

## 1. La théorie complète

### C'est quoi JSON ?

JSON (JavaScript Object Notation) est un format d'échange de données **léger et lisible** par les humains et les machines. C'est le standard universel pour les APIs web.

```json
{
    "id": 1,
    "prenom": "Jack",
    "email": "jack@test.com",
    "actif": true,
    "age": 34,
    "tags": ["php", "mysql", "javascript"],
    "adresse": {
        "ville": "Bruxelles",
        "pays": "Belgique"
    },
    "conjoint": null
}
```

### **`json_encode()` — PHP vers JSON**

```php
<?php
// Tableau associatif → JSON
$data = [
    'id'     => 1,
    'prenom' => 'Jack',
    'actif'  => true,
    'tags'   => ['php', 'mysql'],
];

$json = json_encode($data);
echo $json;
// {"id":1,"prenom":"Jack","actif":true,"tags":["php","mysql"]}

// Options utiles
$json = json_encode($data,
    JSON_PRETTY_PRINT      |   // Formatage indenté
    JSON_UNESCAPED_UNICODE |   // Garde les accents (pas \u00e9)
    JSON_UNESCAPED_SLASHES     // Garde les / (pas \/)
);
/*
{
    "id": 1,
    "prenom": "Jack",
    "actif": true,
    "tags": ["php", "mysql"]
}
*/

// Vérifier les erreurs
$json = json_encode($data);
if (json_last_error() !== JSON_ERROR_NONE) {
    echo "Erreur : " . json_last_error_msg();
}
```

### **`json_decode()` — JSON vers PHP**

```php
<?php
$json = '{"id":1,"prenom":"Jack","actif":true,"tags":["php","mysql"]}';

// Décoder en tableau associatif — 2ème param = true
$data = json_decode($json, true);

echo $data['prenom'];      // "Jack"
echo $data['actif'];       // 1 (true)
var_dump($data['tags']);    // array(2) { "php", "mysql" }

// Sans 2ème param → objet stdClass
$obj = json_decode($json);
echo $obj->prenom;         // "Jack"
echo $obj->actif;          // 1

// Toujours vérifier le résultat
$data = json_decode($json, true);
if ($data === null && json_last_error() !== JSON_ERROR_NONE) {
    throw new \RuntimeException("JSON invalide : " . json_last_error_msg());
}
```

### **Correspondance des types PHP ↔ JSON**

```php
// PHP → JSON
null      → null
true      → true
false     → false
int       → number (42)
float     → number (3.14)
string    → string ("texte")
array[]   → array ([1, 2, 3])
array{}   → object ({"cle": "valeur"})

// JSON → PHP (avec true = tableau associatif)
null      → null
true      → true
false     → false
number    → int ou float selon la valeur
string    → string
array     → array indexé
object    → array associatif (si 2ème param = true)
```

### **Sauvegarder et lire du JSON dans un fichier**

```php
// Lire un fichier JSON
function lireJSON(string $chemin): array
{
    if (!file_exists($chemin)) {
        return [];
    }

    $contenu = file_get_contents($chemin);
    $data    = json_decode($contenu, true);

    if ($data === null) {
        throw new \RuntimeException("Fichier JSON invalide : {$chemin}");
    }

    return $data;
}

// Écrire dans un fichier JSON
function ecrireJSON(string $chemin, array $data): void
{
    $json = json_encode(
        $data,
        JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES
    );

    if ($json === false) {
        throw new \RuntimeException("Impossible d'encoder en JSON");
    }

    file_put_contents($chemin, $json);
}

// Utilisation
$config = lireJSON(__DIR__ . '/config.json');
$config['version'] = '2.0';
ecrireJSON(__DIR__ . '/config.json', $config);
```

### **Construire une réponse API propre**

```php
<?php
declare(strict_types=1);

/**
 * Structure standard d'une réponse API
 * Toujours le même format — facilite le travail côté client
 */

// Succès
function reponseSucces(mixed $data, string $message = '', int $code = 200): void
{
    repondreJSON([
        'succes'    => true,
        'message'   => $message,
        'data'      => $data,
        'timestamp' => time(),
    ], $code);
}

// Erreur
function reponseErreur(string $message, int $code = 400, array $details = []): void
{
    $payload = [
        'succes'    => false,
        'message'   => $message,
        'timestamp' => time(),
    ];

    if (!empty($details)) {
        $payload['details'] = $details;
    }

    repondreJSON($payload, $code);
}

// Utilisation
reponseSucces(['id' => 1, 'titre' => 'Mon article'], 'Article créé', 201);
/*
{
    "succes": true,
    "message": "Article créé",
    "data": {"id": 1, "titre": "Mon article"},
    "timestamp": 1742860800
}
*/

reponseErreur('Validation échouée', 422, ['titre' => 'Le titre est obligatoire']);
/*
{
    "succes": false,
    "message": "Validation échouée",
    "timestamp": 1742860800,
    "details": {"titre": "Le titre est obligatoire"}
}
*/
```

### **Consommer une API externe**

```php
<?php
declare(strict_types=1);

/**
 * Fait une requête HTTP vers une API externe
 * Utilise file_get_contents avec un contexte HTTP
 *
 * @param string $url     URL de l'API
 * @param string $methode GET, POST, PUT, DELETE
 * @param array  $data    Données à envoyer (pour POST/PUT)
 * @param array  $headers Headers supplémentaires
 * @return array          Réponse décodée
 */
function appelAPI(
    string $url,
    string $methode  = 'GET',
    array  $data     = [],
    array  $headers  = []
): array {

    $optionsHeaders = array_merge([
        'Content-Type: application/json',
        'Accept: application/json',
    ], $headers);

    $options = [
        'http' => [
            'method'        => $methode,
            'header'        => implode("\r\n", $optionsHeaders),
            'ignore_errors' => true,   // Ne pas lever d'exception sur 4xx/5xx
        ]
    ];

    if (!empty($data) && in_array($methode, ['POST', 'PUT', 'PATCH'], true)) {
        $options['http']['content'] = json_encode($data);
    }

    $contexte  = stream_context_create($options);
    $reponse   = file_get_contents($url, false, $contexte);

    if ($reponse === false) {
        throw new \RuntimeException("Impossible de contacter l'API : {$url}");
    }

    $decoded = json_decode($reponse, true);

    if ($decoded === null) {
        throw new \RuntimeException("Réponse API invalide");
    }

    return $decoded;
}

// Exemple — appeler une API météo fictive
try {
    $meteo = appelAPI('https://api.exemple.com/meteo?ville=bruxelles');
    echo "Température : " . $meteo['temperature'] . "°C";
} catch (\RuntimeException $e) {
    echo "Erreur API : " . $e->getMessage();
}
```

```php
/**

 * get('pagination.articles_par_page') → 10
```

## **3. Bonnes pratiques et erreurs à éviter**

```php
// ❌ Ne pas vérifier les erreurs de json_decode
$data = json_decode($json, true);
echo $data['prenom'];   // Peut être null si JSON invalide !

// ✅ Toujours vérifier
$data = json_decode($json, true);
if (!is_array($data)) {
    throw new \RuntimeException("JSON invalide : " . json_last_error_msg());
}

// ❌ Encoder sans options — les accents deviennent des codes Unicode
$json = json_encode(['prenom' => 'Élodie']);
// {"prenom":"\u00c9lodie"}

// ✅ Toujours JSON_UNESCAPED_UNICODE
$json = json_encode(['prenom' => 'Élodie'], JSON_UNESCAPED_UNICODE);
// {"prenom":"Élodie"}

// ❌ Stocker des données sensibles en JSON côté client
$json = json_encode([
    'user_id'  => 1,
    'role'     => 'admin',
    'password' => $hash   // ← JAMAIS !
]);

// ✅ Stocker uniquement ce qui est public
$json = json_encode([
    'user_id' => 1,
    'prenom'  => 'Jack',
]);

// ❌ Utiliser json_decode sans le 2ème paramètre
$obj = json_decode($json);         // stdClass — accès par ->
$arr = json_decode($json, true);   // array   — accès par []

// ✅ Toujours true pour cohérence avec le reste du code PHP
$data = json_decode($json, true);
```

### **`src/helpers/config.php`**

```php
<?php
declare(strict_types=1);

/**
 * config.php
 * Gestionnaire de configuration JSON
 * Charge app.json une seule fois grâce au cache statique
 */

/**
 * Charge et retourne la configuration complète
 * Pattern cache statique — le fichier n'est lu qu'une seule fois
 *
 * @return array Configuration complète
 * @throws \RuntimeException Si fichier introuvable ou JSON invalide
 */
function getConfig(): array
{
    // Variable statique — persiste entre les appels de la fonction
    // Initialisée à null, remplie au premier appel
    static $cache = null;

    if ($cache !== null) {
        return $cache;
    }

    $chemin = __DIR__ . '/../config/app.json';

    if (!file_exists($chemin)) {
        throw new \RuntimeException(
            "Fichier de configuration introuvable : {$chemin}"
        );
    }

    $contenu = file_get_contents($chemin);

    if ($contenu === false) {
        throw new \RuntimeException(
            "Impossible de lire le fichier de configuration"
        );
    }

    $cache = json_decode($contenu, true);

    if ($cache === null || json_last_error() !== JSON_ERROR_NONE) {
        throw new \RuntimeException(
            "Fichier de configuration JSON invalide : " . json_last_error_msg()
        );
    }

    return $cache;
}

/**
 * Récupère une valeur de configuration avec notation pointée
 *
 * Exemples :
 *   getConfigValeur('app.nom')                    → "Agence Web"
 *   getConfigValeur('pagination.articles_par_page')→ 5
 *   getConfigValeur('upload.types_autorises')      → ["jpg", ...]
 *   getConfigValeur('inexistant', 'defaut')        → "defaut"
 *
 * @param string $cle    Clé en notation pointée (ex: "app.nom")
 * @param mixed  $defaut Valeur par défaut si la clé n'existe pas
 * @return mixed         Valeur trouvée ou valeur par défaut
 */
function getConfigValeur(string $cle, mixed $defaut = null): mixed
{
    $config  = getConfig();
    $parties = explode('.', $cle);

    // On descend dans le tableau niveau par niveau
    $valeur = $config;
    foreach ($parties as $partie) {
        if (!is_array($valeur) || !array_key_exists($partie, $valeur)) {
            return $defaut;
        }
        $valeur = $valeur[$partie];
    }

    return $valeur;
}

/**
 * Vérifie si une clé de configuration existe
 *
 * @param string $cle Clé en notation pointée
 * @return bool       True si la clé existe
 */
function configExiste(string $cle): bool
{
    $config  = getConfig();
    $parties = explode('.', $cle);

    $valeur = $config;
    foreach ($parties as $partie) {
        if (!is_array($valeur) || !array_key_exists($partie, $valeur)) {
            return false;
        }
        $valeur = $valeur[$partie];
    }

    return true;
}
```

### **`src/helpers/api.php`**

```php
<?php
declare(strict_types=1);

/**
 * api.php
 * Helpers pour les réponses API JSON standardisées
 *
 * Structure de toutes les réponses :
 * {
 *   "succes": true|false,
 *   "code": 200,
 *   "message": "...",
 *   "data": {...} | null,
 *   "timestamp": 1742860800
 * }
 */

// ============================================================
// RÉPONSES STANDARD
// ============================================================

/**
 * Envoie une réponse JSON de succès
 * Utilisée pour toutes les opérations réussies
 *
 * @param mixed  $data    Données à retourner (tableau, objet, null)
 * @param string $message Message de succès
 * @param int    $code    Code HTTP (200 par défaut, 201 pour création)
 */
function reponseSucces(
    mixed  $data    = null,
    string $message = 'OK',
    int    $code    = 200
): void {
    repondreJSON([
        'succes'    => true,
        'code'      => $code,
        'message'   => $message,
        'data'      => $data,
        'timestamp' => time(),
    ], $code);
}

/**
 * Envoie une réponse JSON d'erreur
 * Utilisée pour toutes les erreurs (validation, not found, server error)
 *
 * @param string $message Message d'erreur principal
 * @param int    $code    Code HTTP (400, 404, 422, 500...)
 * @param array  $details Détails supplémentaires (erreurs de validation)
 */
function reponseErreur(
    string $message,
    int    $code    = 400,
    array  $details = []
): void {
    $payload = [
        'succes'    => false,
        'code'      => $code,
        'message'   => $message,
        'timestamp' => time(),
    ];

    // Ajouter les détails seulement s'ils existent
    // Utile pour afficher les erreurs de validation champ par champ
    if (!empty($details)) {
        $payload['details'] = $details;
    }

    repondreJSON($payload, $code);
}

/**
 * Envoie une réponse JSON avec pagination
 * Structure standardisée pour toutes les listes paginées
 *
 * @param array $items   Éléments de la page courante
 * @param int   $total   Nombre total d'éléments (toutes pages)
 * @param int   $page    Page actuelle
 * @param int   $parPage Nombre d'éléments par page
 */
function reponseAvecPagination(
    array $items,
    int   $total,
    int   $page,
    int   $parPage
): void {
    $totalPages = $parPage > 0 ? (int)ceil($total / $parPage) : 1;

    repondreJSON([
        'succes'    => true,
        'code'      => 200,
        'timestamp' => time(),
        'data'      => $items,
        'pagination' => [
            'total'          => $total,
            'page_actuelle'  => $page,
            'par_page'       => $parPage,
            'total_pages'    => $totalPages,
            'a_suivante'     => $page < $totalPages,
            'a_precedente'   => $page > 1,
            'debut'          => (($page - 1) * $parPage) + 1,
            'fin'            => min($page * $parPage, $total),
        ],
    ]);
}

// ============================================================
// HELPERS DE REQUÊTE
// ============================================================

/**
 * Lit et décode le body JSON d'une requête HTTP
 * Utilisé pour les requêtes POST/PUT avec Content-Type: application/json
 *
 * @return array Body décodé en tableau associatif
 * @throws \RuntimeException Si le body est vide ou invalide
 */
function lireBodyJSON(): array
{
    $body = file_get_contents('php://input');

    if (empty($body)) {
        throw new \RuntimeException("Body de requête vide");
    }

    $data = json_decode($body, true);

    if ($data === null || json_last_error() !== JSON_ERROR_NONE) {
        throw new \RuntimeException(
            "Body JSON invalide : " . json_last_error_msg()
        );
    }

    return $data;
}

/**
 * Retourne la méthode HTTP de la requête courante
 *
 * @return string GET, POST, PUT, DELETE, PATCH...
 */
function getMethodeHTTP(): string
{
    return strtoupper($_SERVER['REQUEST_METHOD'] ?? 'GET');
}

/**
 * Vérifie si la requête attend une réponse JSON
 * Basé sur le header Accept
 *
 * @return bool True si le client attend du JSON
 */
function demandeJSON(): bool
{
    $accept = $_SERVER['HTTP_ACCEPT'] ?? '';
    return str_contains($accept, 'application/json');
}
```

```php
    // Compter le total
    $stmtCount = $pdo->prepare("SELECT COUNT(*) as total FROM `{$table}`");
    $stmtCount->execute();
    $total = (int)$stmtCount->fetch()['total'];

    // Récupérer la page
    $stmt = $pdo->prepare("
        SELECT * FROM `{$table}`
        ORDER BY `{$orderBy}` DESC
        LIMIT :limite OFFSET :offset
    ");
    $stmt->bindValue(':limite',  $parPage, PDO::PARAM_INT);
    $stmt->bindValue(':offset',  $offset,  PDO::PARAM_INT);
    $stmt->execute();

    return [
        'data'          => $stmt->fetchAll(),
        'total'         => $total,
        'pages'         => (int)ceil($total / $parPage),
        'page_actuelle' => $page,
        'par_page'      => $parPage,
    ];
}

/**
 * Récupère les enregistrements filtrés par colonne
 */
function fetchWhere(
    string $table,
    string $colonne,
    mixed  $valeur,
    string $orderBy = 'id'
): array {
    validerNomTable($table);
    validerNomTable($colonne);
    validerNomTable($orderBy);

    $pdo  = getConnexion();
    $stmt = $pdo->prepare("
        SELECT * FROM `{$table}`
        WHERE `{$colonne}` = ?
        ORDER BY `{$orderBy}` DESC
    ");
    $stmt->execute([$valeur]);
    return $stmt->fetchAll();
}

/**
 * Récupère un enregistrement par ID
 */
function fetchById(string $table, int $id): array
{
    validerNomTable($table);

    $pdo  = getConnexion();
    $stmt = $pdo->prepare("SELECT * FROM `{$table}` WHERE id = ?");
    $stmt->execute([$id]);

    $enregistrement = $stmt->fetch();

    if (!$enregistrement) {
        throw new NotFoundException(
            "Enregistrement ID {$id} introuvable dans `{$table}`",
            404
        );
    }

    return $enregistrement;
}

/**
 * Insère un enregistrement
 */
function insert(string $table, array $donnees): int
{
    validerNomTable($table);

    $pdo          = getConnexion();
    $colonnes     = implode(', ', array_map(fn($c) => "`{$c}`", array_keys($donnees)));
    $placeholders = implode(', ', array_fill(0, count($donnees), '?'));

    $stmt = $pdo->prepare("
        INSERT INTO `{$table}` ({$colonnes})
        VALUES ({$placeholders})
    ");
    $stmt->execute(array_values($donnees));

    return (int)$pdo->lastInsertId();
}

/**
 * Met à jour un enregistrement
 */
function update(string $table, int $id, array $donnees): int
{
    validerNomTable($table);

    $pdo = getConnexion();
    $set = implode(', ', array_map(
        fn(string $col): string => "`{$col}` = ?",
        array_keys($donnees)
    ));

    $valeurs   = array_values($donnees);
    $valeurs[] = $id;

    $stmt = $pdo->prepare("UPDATE `{$table}` SET {$set} WHERE id = ?");
    $stmt->execute($valeurs);

    $lignes = $stmt->rowCount();

    if ($lignes === 0) {
        throw new NotFoundException(
            "Enregistrement ID {$id} introuvable dans `{$table}`",
            404
        );
    }

    return $lignes;
}

/**
 * Supprime un enregistrement
 */
function delete(string $table, int $id): int
{
    validerNomTable($table);

    $pdo  = getConnexion();
    $stmt = $pdo->prepare("DELETE FROM `{$table}` WHERE id = ?");
    $stmt->execute([$id]);

    $lignes = $stmt->rowCount();

    if ($lignes === 0) {
        throw new NotFoundException(
            "Enregistrement ID {$id} introuvable dans `{$table}`",
            404
        );
    }

    return $lignes;
}

/**
 * Recherche des articles par mot-clé
 */
function rechercherArticles(string $motCle): array
{
    $pdo  = getConnexion();
    $like = '%' . $motCle . '%';

    $stmt = $pdo->prepare("
        SELECT * FROM articles
        WHERE titre   LIKE :motcle
        OR    contenu LIKE :motcle
        ORDER BY created_at DESC
    ");
    $stmt->execute([':motcle' => $like]);
    return $stmt->fetchAll();
}
```

### **`public/api.php` — mise à jour complète**

```php
<?php
declare(strict_types=1);

/**
 * api.php — API JSON complète
 *
 * GET    /api.php?ressource=articles           → liste paginée
 * GET    /api.php?ressource=articles&id=1      → un article
 * POST   /api.php?ressource=articles           → créer
 * PUT    /api.php?ressource=articles&id=1      → modifier
 * DELETE /api.php?ressource=articles&id=1      → supprimer
 */

require_once __DIR__ . '/../src/config/config.php';
require_once __DIR__ . '/../src/helpers/errors.php';
require_once __DIR__ . '/../src/helpers/security.php';
require_once __DIR__ . '/../src/helpers/validation.php';
require_once __DIR__ . '/../src/helpers/functions.php';
require_once __DIR__ . '/../src/helpers/http.php';
require_once __DIR__ . '/../src/helpers/api.php';
require_once __DIR__ . '/../src/helpers/config.php';
require_once __DIR__ . '/../src/config/database.php';
require_once __DIR__ . '/../src/helpers/crud.php';

// Toutes les réponses sont en JSON
header('Content-Type: application/json; charset=UTF-8');
header('Cache-Control: no-store');

// Ressources et méthode
$ressource = preg_replace('/[^a-z_]/', '', $_GET['ressource'] ?? '');
$id        = (int)($_GET['id'] ?? 0);
$methode   = getMethodeHTTP();

// Ressources autorisées
$ressourcesAutorisees = ['articles'];

if (!in_array($ressource, $ressourcesAutorisees, true)) {
    reponseErreur("Ressource inconnue : {$ressource}", 404);
}

// ============================================================
// ROUTAGE
// ============================================================

try {

    // ── GET — liste paginée ou détail ────────────────────────
    if ($methode === 'GET') {

        if ($id > 0) {
            // GET ?ressource=articles&id=1
            $article = fetchById($ressource, $id);
            reponseSucces($article, 'Article trouvé');

        } else {
            // GET ?ressource=articles — liste paginée
            $page    = max(1, (int)($_GET['p']        ?? 1));
            $parPage = min(50, max(1, (int)($_GET['par_page']
                ?? getConfigValeur('pagination.articles_par_page', 5))));

            $result = fetchAll($ressource, 'created_at', $page, $parPage);

            reponseAvecPagination(
                $result['data'],
                $result['total'],
                $result['page_actuelle'],
                $result['par_page']
            );
        }
    }

    // ── POST — créer ─────────────────────────────────────────
    if ($methode === 'POST') {

        // Lire le body JSON
        try {
            $body = lireBodyJSON();
        } catch (\RuntimeException $e) {
            reponseErreur("Body JSON invalide ou manquant", 400);
        }

        // Validation avec validerDonnees()
        $regles = [
            'titre' => [
                'required'    => true,
                'min'         => 3,
                'max'         => 255,
                'message'     => "Le titre est obligatoire",
                'message_min' => "Le titre doit faire au moins 3 caractères",
                'message_max' => "Le titre ne doit pas dépasser 255 caractères",
            ],
            'contenu' => [
                'required' => true,
                'message'  => "Le contenu est obligatoire",
            ],
            'auteur' => [
                'required' => true,
                'min'      => 2,
                'message'  => "L'auteur est obligatoire",
            ],
            'statut' => [
                'required' => false,
                'in'       => ['publie', 'brouillon'],
                'message'  => "Le statut doit être 'publie' ou 'brouillon'",
            ],
        ];

        $validation = validerDonnees($body, $regles);

        if (!$validation['valide']) {
            reponseErreur(
                "Validation échouée",
                422,
                $validation['erreurs']
            );
        }

        $id = insert($ressource, [
            'titre'   => trim($body['titre']),
            'contenu' => trim($body['contenu']),
            'auteur'  => trim($body['auteur']),
            'statut'  => $body['statut'] ?? 'brouillon',
        ]);

        $articleCree = fetchById($ressource, $id);
        reponseSucces($articleCree, 'Article créé avec succès', 201);
    }

    // ── PUT — modifier ───────────────────────────────────────
    if ($methode === 'PUT') {

        if ($id <= 0) {
            reponseErreur("ID manquant ou invalide", 400);
        }

        try {
            $body = lireBodyJSON();
        } catch (\RuntimeException $e) {
            reponseErreur("Body JSON invalide", 400);
        }

        // Récupérer l'article existant
        $existant = fetchById($ressource, $id);

        // Merge — ne modifier que les champs fournis
        $donnees = [
            'titre'   => trim($body['titre']   ?? $existant['titre']),
            'contenu' => trim($body['contenu'] ?? $existant['contenu']),
            'auteur'  => trim($body['auteur']  ?? $existant['auteur']),
            'statut'  => $body['statut']       ?? $existant['statut'],
        ];

        if (!in_array($donnees['statut'], ['publie', 'brouillon'], true)) {
            $donnees['statut'] = 'brouillon';
        }

        update($ressource, $id, $donnees);

        $articleModifie = fetchById($ressource, $id);
        reponseSucces($articleModifie, "Article {$id} modifié avec succès");
    }

    // ── DELETE — supprimer ───────────────────────────────────
    if ($methode === 'DELETE') {

        if ($id <= 0) {
            reponseErreur("ID manquant ou invalide", 400);
        }

        // Vérifier que l'article existe avant de supprimer
        $existant = fetchById($ressource, $id);

        delete($ressource, $id);

        reponseSucces(
            ['id_supprime' => $id],
            "Article \"{$existant['titre']}\" supprimé avec succès"
        );
    }

    // Méthode non supportée
    reponseErreur("Méthode {$methode} non supportée", 405);

} catch (NotFoundException $e) {
    reponseErreur($e->getMessage(), 404);
} catch (ValidationException $e) {
    reponseErreur($e->getMessage(), 400);
} catch (PDOException $e) {
    loggerErreur($e);
    reponseErreur('Erreur base de données', 500);
} catch (\RuntimeException $e) {
    loggerErreur($e);
    reponseErreur($e->getMessage(), 400);
} catch (Exception $e) {
    loggerErreur($e);
    reponseErreur('Erreur serveur inattendue', 500);
}
```
