# 🔹 Étape 19 — Gestion des fichiers

## 1. La théorie complète

### Les fonctions de base

```php
<?php
// ── LIRE UN FICHIER ──────────────────────────────────────────

// file_get_contents — lit tout le fichier en une chaîne
$contenu = file_get_contents('/chemin/vers/fichier.txt');

// file — lit ligne par ligne dans un tableau
$lignes = file('/chemin/vers/fichier.txt', FILE_IGNORE_NEW_LINES);

// fopen / fread / fclose — lecture manuelle (gros fichiers)
$handle = fopen('/chemin/vers/fichier.txt', 'r');
if ($handle) {
    while (($ligne = fgets($handle)) !== false) {
        echo $ligne;
    }
    fclose($handle);
}

// ── ÉCRIRE UN FICHIER ────────────────────────────────────────

// file_put_contents — écrit une chaîne dans un fichier
file_put_contents('/chemin/vers/fichier.txt', 'Contenu');

// FILE_APPEND — ajouter à la fin sans écraser
file_put_contents('/chemin/vers/fichier.txt', "Nouvelle ligne\n", FILE_APPEND);

// LOCK_EX — verrouiller le fichier pendant l'écriture
file_put_contents('/chemin/vers/fichier.txt', $contenu, LOCK_EX);

// fopen / fwrite / fclose — écriture manuelle
$handle = fopen('/chemin/vers/fichier.txt', 'w');  // 'w' = écriture (écrase)
if ($handle) {
    fwrite($handle, "Contenu ligne 1\n");
    fwrite($handle, "Contenu ligne 2\n");
    fclose($handle);
}
```

### **Les modes d'ouverture de fichier**

```php
// Mode 'r'  — Lecture seule — pointeur au début
// Mode 'w'  — Écriture seule — crée ou ÉCRASE le fichier
// Mode 'a'  — Écriture seule — crée ou AJOUTE à la fin
// Mode 'r+' — Lecture + écriture — pointeur au début
// Mode 'w+' — Lecture + écriture — crée ou ÉCRASE
// Mode 'a+' — Lecture + écriture — crée ou AJOUTE à la fin
```

### **Upload de fichiers**

```php
// Formulaire HTML nécessaire
// <form method="POST" enctype="multipart/form-data">
// <input type="file" name="fichier">

// $_FILES['fichier'] contient :
// [
//   'name'     => 'photo.jpg',        // Nom original
//   'type'     => 'image/jpeg',        // MIME type déclaré (NON FIABLE !)
//   'tmp_name' => '/tmp/php7f3x2k',   // Chemin temporaire
//   'error'    => 0,                   // 0 = UPLOAD_ERR_OK
//   'size'     => 245760              // Taille en octets
// ]

// Codes d'erreur upload
// UPLOAD_ERR_OK         = 0  — Pas d'erreur
// UPLOAD_ERR_INI_SIZE   = 1  — Dépasse upload_max_filesize
// UPLOAD_ERR_FORM_SIZE  = 2  — Dépasse MAX_FILE_SIZE du formulaire
// UPLOAD_ERR_PARTIAL    = 3  — Upload partiel
// UPLOAD_ERR_NO_FILE    = 4  — Aucun fichier envoyé
// UPLOAD_ERR_NO_TMP_DIR = 6  — Dossier temporaire manquant
// UPLOAD_ERR_CANT_WRITE = 7  — Impossible d'écrire sur le disque

// ── Validation d'un upload ───────────────────────────────────

function validerUpload(array $fichier, array $options = []): array
{
    $erreurs       = [];
    $tailleMaxOctets = ($options['taille_max_mo'] ?? 5) * 1024 * 1024;
    $typesAutorises  = $options['types_autorises'] ?? ['jpg', 'jpeg', 'png', 'pdf'];

    // 1. Vérifier l'erreur d'upload
    if ($fichier['error'] !== UPLOAD_ERR_OK) {
        $messages = [
            UPLOAD_ERR_INI_SIZE   => "Fichier trop volumineux (limite serveur)",
            UPLOAD_ERR_FORM_SIZE  => "Fichier trop volumineux",
            UPLOAD_ERR_PARTIAL    => "Upload incomplet",
            UPLOAD_ERR_NO_FILE    => "Aucun fichier sélectionné",
            UPLOAD_ERR_NO_TMP_DIR => "Erreur serveur (dossier temporaire)",
            UPLOAD_ERR_CANT_WRITE => "Erreur serveur (écriture impossible)",
        ];
        $erreurs[] = $messages[$fichier['error']] ?? "Erreur d'upload inconnue";
        return $erreurs;
    }

    // 2. Vérifier la taille
    if ($fichier['size'] > $tailleMaxOctets) {
        $erreurs[] = "Le fichier ne doit pas dépasser {$options['taille_max_mo']} Mo";
    }

    // 3. Vérifier l'extension — depuis le nom du fichier
    $extension = strtolower(pathinfo($fichier['name'], PATHINFO_EXTENSION));
    if (!in_array($extension, $typesAutorises, true)) {
        $erreurs[] = "Type de fichier non autorisé. Autorisés : "
            . implode(', ', $typesAutorises);
    }

    // 4. Vérifier le MIME type réel — depuis le fichier lui-même
    // finfo est plus fiable que $_FILES['type'] qui vient du client
    $finfo    = new finfo(FILEINFO_MIME_TYPE);
    $mimeReel = $finfo->file($fichier['tmp_name']);

    $mimesAutorises = [
        'jpg'  => 'image/jpeg',
        'jpeg' => 'image/jpeg',
        'png'  => 'image/png',
        'gif'  => 'image/gif',
        'pdf'  => 'application/pdf',
        'docx' => 'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
    ];

    if (
        isset($mimesAutorises[$extension]) &&
        $mimeReel !== $mimesAutorises[$extension]
    ) {
        $erreurs[] = "Le contenu du fichier ne correspond pas à son extension";
    }

    return $erreurs;
}

// ── Déplacer l'upload vers le dossier final ──────────────────

function deplacerUpload(array $fichier, string $dossierDestination): string
{
    // Créer le dossier si nécessaire
    if (!is_dir($dossierDestination)) {
        mkdir($dossierDestination, 0755, true);
    }

    // Générer un nom unique — éviter les collisions et les noms malveillants
    $extension   = strtolower(pathinfo($fichier['name'], PATHINFO_EXTENSION));
    $nomUnique   = bin2hex(random_bytes(16)) . '.' . $extension;
    $destination = rtrim($dossierDestination, '/') . '/' . $nomUnique;

    if (!move_uploaded_file($fichier['tmp_name'], $destination)) {
        throw new \RuntimeException("Impossible de déplacer le fichier uploadé");
    }

    return $nomUnique;   // Retourner le nom pour le stocker en DB
}
```

### **Logger dans un fichier**

```php
<?php
declare(strict_types=1);

/**
 * Écrit une entrée dans un fichier de log
 *
 * @param string $message  Message à logger
 * @param string $niveau   debug, info, warning, error
 * @param string $fichierLog Chemin vers le fichier de log
 */
function loggerMessage(
    string $message,
    string $niveau    = 'info',
    string $fichierLog = ''
): void {
    if (empty($fichierLog)) {
        $fichierLog = __DIR__ . '/../../logs/app.log';
    }

    // Créer le dossier logs si nécessaire
    $dossier = dirname($fichierLog);
    if (!is_dir($dossier)) {
        mkdir($dossier, 0755, true);
    }

    $ligne = sprintf(
        "[%s] [%s] %s\n",
        date('Y-m-d H:i:s'),
        strtoupper($niveau),
        $message
    );

    // FILE_APPEND = ajouter à la fin
    // LOCK_EX = verrouiller pendant l'écriture
    file_put_contents($fichierLog, $ligne, FILE_APPEND | LOCK_EX);
}
```

## **2. Exemple concret — système d'upload d'images**

```php
<?php
declare(strict_types=1);

/**
 * upload.php
 * Gestion complète de l'upload d'images pour les articles
 */

const UPLOAD_DIR     = __DIR__ . '/../uploads/';
const UPLOAD_MAX_MO  = 5;
const UPLOAD_TYPES   = ['jpg', 'jpeg', 'png', 'gif', 'webp'];

$erreurs = [];
$succes  = null;

if ($_SERVER['REQUEST_METHOD'] === 'POST') {

    if (!verifierTokenCSRF($_POST['csrf_token'] ?? '')) {
        http_response_code(403);
        die("Requête invalide");
    }

    if (!isset($_FILES['image']) || $_FILES['image']['error'] === UPLOAD_ERR_NO_FILE) {
        $erreurs[] = "Veuillez sélectionner un fichier";
    } else {

        // Valider
        $erreursUpload = validerUpload($_FILES['image'], [
            'taille_max_mo'  => UPLOAD_MAX_MO,
            'types_autorises'=> UPLOAD_TYPES,
        ]);

        if (!empty($erreursUpload)) {
            $erreurs = $erreursUpload;
        } else {

            try {
                // Déplacer vers le dossier final
                $nomFichier = deplacerUpload($_FILES['image'], UPLOAD_DIR);

                // Stocker le nom en base de données
                // insert('images', ['nom' => $nomFichier, 'article_id' => $id]);

                $succes = "Image uploadée : {$nomFichier}";
                loggerMessage("Upload réussi : {$nomFichier}", 'info');

            } catch (\RuntimeException $e) {
                loggerErreur($e);
                $erreurs[] = "Erreur lors de l'upload";
            }
        }
    }
}
```

## **3. Bonnes pratiques et erreurs à éviter**

```php

move_uploaded_file($_FILES['fichier']['tmp_name'], 'uploads/' . $nom);
// Un attaquant envoie "shell.php" → exécution de code PHP !

// ✅ Toujours générer un nom aléatoire

// ❌ Faire confiance au MIME type du client

if ($type === 'image/jpeg') { }

// ✅ Vérifier le MIME réel avec finfo
$finfo    = new finfo(FILEINFO_MIME_TYPE);

// ❌ Stocker les uploads dans un dossier public sans restriction
// uploads/shell.php accessible via url.com/uploads/shell.php !

// ✅ Stocker hors du dossier public OU avec .htaccess
// Option 1 : dossier hors de public/
// /var/www/agence/uploads/   (pas dans public/)

//     Require all denied
// </Files>

// Deux requêtes simultanées peuvent corrompre le fichier
```

## 4. Ce que font les seniors 👨‍💻

Les seniors créent un service d'upload réutilisable et sécurisé :

```php
<?php
// src/helpers/upload.php
declare(strict_types=1);

/**
 * Service d'upload centralisé
 * Toute la logique d'upload au même endroit
 */

/**
 * Traite un upload de fichier de bout en bout
 * Valide, sécurise et déplace le fichier
 *
 * @param array  $fichier   $_FILES['champ']
 * @param string $dossier   Sous-dossier dans uploads/ (ex: 'images', 'docs')
 * @param array  $options   taille_max_mo, types_autorises
 * @return array            ['succes' => bool, 'nom' => '...', 'erreurs' => []]
 */
function traiterUpload(array $fichier, string $dossier, array $options = []): array
{
    // Options avec valeurs par défaut depuis la config
    $tailleMax    = $options['taille_max_mo']  ?? getConfigValeur('upload.taille_max_mo', 5);
    $typesOk      = $options['types_autorises']?? getConfigValeur('upload.types_autorises', ['jpg','png','pdf']);

    // 1. Valider
    $erreurs = validerUpload($fichier, [
        'taille_max_mo'  => $tailleMax,
        'types_autorises'=> $typesOk,
    ]);

    if (!empty($erreurs)) {
        return ['succes' => false, 'nom' => '', 'erreurs' => $erreurs];
    }

    // 2. Construire le chemin de destination
    $dossierBase = __DIR__ . '/../../uploads/' . trim($dossier, '/');

    // 3. Déplacer
    try {
        $nomFichier = deplacerUpload($fichier, $dossierBase);

        loggerMessage(
            "Upload : {$nomFichier} dans {$dossier} ({$fichier['size']} octets)",
            'info'
        );

        return ['succes' => true, 'nom' => $nomFichier, 'erreurs' => []];

    } catch (\RuntimeException $e) {
        loggerErreur($e);
        return ['succes' => false, 'nom' => '', 'erreurs' => [$e->getMessage()]];
    }
}

/**
 * Supprime un fichier uploadé
 *
 * @param string $nom     Nom du fichier
 * @param string $dossier Sous-dossier dans uploads/
 * @return bool           True si supprimé
 */
function supprimerFichierUpload(string $nom, string $dossier): bool
{
    // Sécurité — le nom ne doit pas contenir de chemin
    if ($nom !== basename($nom)) {
        loggerMessage("Tentative de traversée de chemin : {$nom}", 'warning');
        return false;
    }

    $chemin = __DIR__ . '/../../uploads/' . trim($dossier, '/') . '/' . $nom;

    if (!file_exists($chemin)) {
        return false;
    }

    $resultat = unlink($chemin);

    if ($resultat) {
        loggerMessage("Fichier supprimé : {$nom}", 'info');
    }

    return $resultat;
}

/**
 * Retourne l'URL publique d'un fichier uploadé
 *
 * @param string $nom     Nom du fichier
 * @param string $dossier Sous-dossier
 * @return string         URL complète
 */
function urlFichierUpload(string $nom, string $dossier): string
{
    return APP_URL . '/uploads/' . trim($dossier, '/') . '/' . basename($nom);
}
```
